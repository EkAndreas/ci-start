#!/bin/bash
if [ -z "$1" ]; then
  echo "`basename $0` {start|stop}"
  exit
fi

case "$1" in
start)
  /usr/bin/Xvfb :99 -ac -screen 0 1024x768x8 &
;;
stop)
  killall Xvfb
;;
esac

sudo chmod 755 /etc/init.d/xvfb

sudo update-rc.d xvfb defaults 10
